A - shiki

# define _CRT_SECURE_NO_WARNINGS
# include <cctype>
# include <cstdio>
# include <cstring>

namespace Main {
	namespace Sourse {
		typedef long long signed int lld;
		typedef unsigned int uint;
		typedef long long unsigned int llu;
		static inline char getc() {
			char t;
			while (isspace(t = getchar()));
			return t;
		}
		static inline const void gets(char* s) {
			char t;
			while (isspace(t = getchar()));
			while (*s++ = t, islower(t = getchar()));
		}
		static inline uint read() {
			char t;
			while (isspace(t = getchar()));
			uint r(0);
			while (r = r * 10 + (t - '0'), isdigit(t = getchar()));
			return r;
		}
		static inline const void read(uint& a) { a = read(); }
		static inline const lld max(const lld a, const lld b) { return a < b ? b : a; }
		static inline const void amax(lld& a, const lld b) { a = max(a, b); }
	}
	using namespace Sourse;
	enum { N = (const uint)1e5, M = (const uint)1e5 };
	static uint m, n;
	static llu l['z' + 1]['z' + 1];
	static char s[N + 1];
	static lld f[N + 1]['z' + 1];
	static inline const void begin() {
		freopen("shiki.in", "r", stdin), freopen("shiki.out", "w", stdout);
		read(n), gets(s), read(m);
		for (register uint i(0); i < m; ++i) {
			const char t(getc()), u(getc()); const uint k(read());
			l[t][u] += k;
		}
	}
	static inline const void end() {
		printf("%llu\n", f[n][0]);
		fclose(stdin), fclose(stdout);
	}
	static inline const void solve() {
		memset(f, 0x80, sizeof f), f[0][s[0]] = 0;
		for (register uint i(1); i <= n; ++i) {
			for (register uint j('a'); j <= 'z'; ++j) f[i][j] = f[i - 1][j];
			f[i][0] = f[i - 1][0];
			for (register uint j('a'); j <= 'z'; ++j) amax(f[i][s[i]], f[i - 1][j] + l[j][s[i]]);
			f[i][s[i]] = max(f[i][s[i]], f[i - 1][0] + l[0][s[i]]);
		}
	}
	static inline const void main() {
		begin();
		solve();
		end();
	}
}

signed int main() { Main::main(); return 0; }

B - pokemon

# define _CRT_SECURE_NO_WARNIGNS
# include <cctype>
# include <cstdio>

namespace Main {
	namespace Sourse {
		typedef short signed int hd;
		typedef short unsigned int hu;
		typedef unsigned int uint;
		static inline char getc() { char t; while (isspace(t = getchar())); return t; }
		static inline uint read() {
			char t;
			while (isspace(t = getchar()));
			uint r(0);
			while (r = r * 10 + (t - '0'), isdigit(t = getchar()));
			return r;
		}
		static inline const void read(uint& a) { a = read(); }
		static inline const void write(uint a) {
			if (not a) return (const void)puts("0");
			uint t(1);
			while (t <= a) t *= 10;
			while (t /= 10) putchar(a / t % 10 + '0');
			putchar('\n');
		}
		static inline const uint min(const uint a, const uint b) { return b < a ? b : a; }
		static inline const void amin(uint& a, const uint b) { a = min(a, b); }
	}
	using namespace Sourse;
	enum { N = (const uint)5e5, M = (const uint)5e5, NM = (const uint)5e5 };
	enum { N2M2 = NM + N * 2 + M * 2 + 4 };
	static const hd d[4][2]{ { -1, 0 }, { 0, -1 }, { 1, 0 }, { 0, 1 } };
	static uint n, m;
	static bool _map[N2M2];
	static uint q;
	static uint _dis[N2M2 * 4];
	static bool _vis[N2M2 * 4];
	static uint sx, sy;
	static hu sd;
	static uint step;
	static inline const hu hash(const char d)
	{ switch (d) { case 'U': return 0; case 'D': return 2; case 'L': return 1; case 'R': return 3; } }
	static inline const uint hash(const uint x, const uint y) { return x * (m + 2) + y; }
	static inline const uint hashd(const uint x, const uint y, const hu d) { return hash(x, y) * 4 + d; }
	static inline bool& map(const uint x, const uint y) { return _map[hash(x, y)]; }
	static inline uint& dis(const uint x, const uint y, const hu d) { return _dis[hashd(x, y, d)]; }
	static inline bool& vis(const uint x, const uint y, const hu d) { return _vis[hashd(x, y, d)]; }
	static inline const void correct(const uint x, const uint y, hu& dir) {
		for (register hu i(0); i < 4; ++i)
			if (map(x + d[(dir + i + 3) % 4][0], y + d[(dir + i + 3) % 4][1]))
				return (const void)(dir = (dir + i + 3) % 4);
	}
	static inline const void move(uint& x, uint& y, hu& dir) { x += d[dir][0], y += d[dir][1], correct(x, y, dir); }
	static inline const void search() {
		static uint x, y;
		for (x = 1, y = 1; !map(x, y); y <= m ? ++y : (++x, y = 0));
		static hu d(0); correct(x, y, d);
		sx = x, sy = y, sd = d;
		while (vis(x, y, d) = true, dis(x, y, d) = ++step, move(x, y, d), x != sx or y != sy or d != sd);
	}
	static inline const void begin() {
		freopen("pokemon.in", "r", stdin), freopen("pokemon.out", "w", stdout);
		read(n), read(m);
		for (uint i(1); i <= n; ++i) for (uint j(1); j <= m; ++j) map(i, j) = getc() == '.';
		search();
	}
	static inline const void end() {
		fclose(stdin), fclose(stdout);
	}
	static inline const void solve() {
		const uint ax(read()), ay(read()), bx(read()), by(read());
		const hu dir(hash(getc()));
		uint ans(compl 0u);
		for (hu i(0); i < 4; ++i) if (vis(bx, by, i)) amin(ans, (dis(bx, by, i) + step - dis(ax, ay, dir)) % step);
		write(ans);
	}
	static inline const void main() {
		begin();
		read(q);
		for (register uint i(0); i < q; ++i) solve();
		end();
	}
}

signed int main() { Main::main(); return 0; }

C - matrix

# define _CRT_SECURE_NO_WARNINGS
# include <cstdio>
# include <cstring>

namespace Main {
	namespace Sourse {
		typedef short signed int hd;
		typedef short unsigned int hu;
		typedef unsigned int uint;
		static inline const hu max(const hu a, const hu b) { return a < b ? b : a; }
		static inline const void amax(hu& a, const hu b) { a = max(a, b); }
	}
	using namespace Sourse;
	enum { NM = 40000, N = NM, M = NM, A = 40000 };
	static const hd d[4][2]{ { -1, 0 }, { 0, -1 }, { 0, 1 }, { 1, 0 } };
	static hu n, m;
	static hu _mat[NM + N + M + 4];
	static hu _ans[NM + N + M + 4];
	static hu k;
	static hu res(1);
	static inline const uint hash(const hu x, const hu y) { return x * (m + 2) + y; }
	static inline hu& mat(const hu x, const hu y) { return _mat[hash(x, y)]; }
	static inline hu& ans(const hu x, const hu y) { return _ans[hash(x, y)]; }
	static inline const hu search(const hu x, const hu y) {
		if (ans(x, y)) return ans(x, y);
		ans(x, y) = 1;
		for (hu i(0); i < 4; ++i)
			if (mat(x + d[i][0], y + d[i][1]) == 1u * mat(x, y) * k)
				amax(ans(x, y), search(x + d[i][0], y + d[i][1]) + 1);
		return ans(x, y);
	}
	static inline const void main() {
		freopen("matrix.in", "r", stdin), freopen("matrix.out", "w", stdout);
		scanf("%hu%hu", &n, &m);
		bool f(false);
		if (!n or !m) { puts("0"); goto end; }
		for (hu i(1); i <= n; ++i) for (hu j(1); j <= m; ++j) scanf("%hu", &mat(i, j));
		for (hu i(1); i <= n; ++i) for (hu j(2); j <= m; ++j) f = f || mat(i, j) == mat(i, j - 1);
		for (hu i(2); i <= n; ++i) for (hu j(1); j <= m; ++j) f = f || mat(i, j) == mat(i - 1, j);
		if (f) { puts("-1"); goto end; }
		for (hu i(1); i <= n; ++i) for (hu j(2); j <= m; ++j) if (not(mat(i, j) % mat(i, j - 1))) res = 2;
		for (hu i(1); i <= n; ++i) for (hu j(2); j <= m; ++j) if (not(mat(i, j - 1) % mat(i, j))) res = 2;
		for (hu i(2); i <= n; ++i) for (hu j(1); j <= m; ++j) if (not(mat(i, j) % mat(i - 1, j))) res = 2;
		for (hu i(2); i <= n; ++i) for (hu j(1); j <= m; ++j) if (not(mat(i - 1, j) % mat(i, j))) res = 2;
		for (hu i(2); 1u * i * i <= A; ++i) {
			memset(_ans, 0, sizeof _ans);
			k = i;
			for (hu i(1); i <= n; ++i) for (hu j(1); j <= m; ++j) amax(res, search(i, j));
		}
		printf("%hu\n", res);
	end:fclose(stdin), fclose(stdout);
	}
}

signed int main() { Main::main(); return 0; }

# define _CRT_SECURE_NO_WARNINGS
# include <cstdio>

namespace Main {
    namespace Sourse {
        typedef unsigned int uint;
        typedef long long unsigned int llu;
    }
    using namespace Sourse;
    enum { N = (const uint)1e9, L = (const uint)1e5 };
    static llu fact(1);
    static inline const void main() {
        freopen("temp.txt", "w", stdout);
        printf("%llu", fact);
        for (register uint i(1); i <= N; ++i) {
            uint t(i);
            while (not(t bitand 1)) t >>= 1;
            fact *= t;
            if (not(i % L)) printf(", %lluull", fact);
        }
        fclose(stdout);
    }
}

signed int main() { Main::main(); return 0; }

// multiplication.cpp
# define _CRT_SECURE_NO_WARNINGS
# include <cstdio>
# include <map>
# include <vector>

namespace Main {
	namespace Sourse {
		typedef short unsigned int hu;
		typedef unsigned int uint;
		typedef long long unsigned llu;
		typedef std::vector<llu> vector;
		typedef std::map<llu, vector> map;
		static inline const void putc(const hu x) { putchar(x < 10 ? x + '0' : x - 10 + 'A'); }
		static inline const void write(llu a) {
			if (not a) return (const void)puts("0");
			hu t(60);
			while (not(a >> t)) t -= 4;
			while (putc(a >> t bitand 0xF), (t -= 4) < 64);
			putchar('\n');
		}
	}
	using namespace Sourse;
	enum { N = (const uint)1e9 };
	static hu t;
	static llu ft[N / 100000 + 1]{ ... }; // table
	static inline const void solve() {
		static uint n; scanf("%u", &n);
		llu fact(ft[n / 100000]);
		for (uint i(n - n % 100000 + 1); i <= n; ++i) {
			uint t(i);
			while (not(t & 1)) t >>= 1;
			fact *= t;
		}
		llu cnt(0);
		for (uint pow(2); pow <= n; pow <<= 1) cnt += n / pow;
		write(fact << (cnt % 4));
	}
	static inline const void main() {
		freopen("multiplication.in", "r", stdin), freopen("multiplication.out", "w", stdout);
		scanf("%hu", &t);
		for (hu i(0); i < t; ++i) solve();
		fclose(stdin), fclose(stdout);
	}
}

signed int main() { Main::main(); return 0; }
