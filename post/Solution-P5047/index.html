[Solution] P5047 #Ynoi2019M B

发布于 2021-05-03 | 分类于 Solution | 12分钟 | 1927字数

P5047. Luogu blog.

显然需要先离散化。然后考虑莫队。

第一层莫队有四种转移（ccc 代表所求逆序对数）。

  1. [l,r]→[l′,r](l′⩽l⩽r)[l, r] \rightarrow [l', r] (l' \leqslant l \leqslant r)[l,r]→[l′,r](l′⩽l⩽r)：

    Δc=+∑i=l′l−1∑j=i+1r[aj<ai]=+∑i=l′l−1(∑j=i+1n[aj<ai]−∑j=r+1n[aj<ai])\Delta c = +\sum_{i = l'}^{l - 1} \sum_{j = i + 1}^r [a_j < a_i] = +\sum_{i = l'}^{l - 1}(\sum_{j = i + 1}^n[a_j < a_i] - \sum_{j = r + 1}^n[a_j < a_i])Δc=+∑i=l′l−1​∑j=i+1r​[aj​<ai​]=+∑i=l′l−1​(∑j=i+1n​[aj​<ai​]−∑j=r+1n​[aj​<ai​])。

  2. [l,r]→[l′,r](l⩽l′⩽r)[l, r] \rightarrow [l', r] (l \leqslant l' \leqslant r)[l,r]→[l′,r](l⩽l′⩽r)：

    Δc=−∑i=ll′−1∑j=i+1r[aj<ai]=−∑i=ll′−1(∑j=i+1n[aj<ai]−∑j=r+1n[aj<ai])\Delta c = -\sum_{i = l}^{l' - 1} \sum_{j = i + 1}^r [a_j < a_i] = -\sum_{i = l}^{l' - 1} (\sum_{j = i + 1}^n [a_j < a_i] - \sum_{j = r + 1}^n [a_j < a_i])Δc=−∑i=ll′−1​∑j=i+1r​[aj​<ai​]=−∑i=ll′−1​(∑j=i+1n​[aj​<ai​]−∑j=r+1n​[aj​<ai​])。

  3. [l,r]→[l,r′](l⩽r′⩽r)[l, r] \rightarrow [l, r'] (l \leqslant r' \leqslant r)[l,r]→[l,r′](l⩽r′⩽r)：

    Δc=−∑i=r′+1r∑j=li−1[ai<aj]=−∑i=r′+1r(∑j=1i−1[ai<aj]−∑j=1l−1[ai<aj])\Delta c = - \sum_{i = r' + 1}^r \sum_{j = l}^{i - 1} [a_i < a_j] = - \sum_{i = r' + 1}^r (\sum_{j = 1}^{i - 1} [a_i < a_j] - \sum_{j = 1}^{l - 1} [a_i < a_j])Δc=−∑i=r′+1r​∑j=li−1​[ai​<aj​]=−∑i=r′+1r​(∑j=1i−1​[ai​<aj​]−∑j=1l−1​[ai​<aj​])。

  4. [l,r]→[l,r′](l⩽r⩽r′)[l, r] \rightarrow [l, r'] (l \leqslant r \leqslant r')[l,r]→[l,r′](l⩽r⩽r′)：

    Δc=+∑i=r+1r′∑j=li−1[ai<aj]=+∑i=r+1r′(∑j=1i−1[ai<aj]−∑j=1l−1[ai<aj])\Delta c = + \sum_{i = r + 1}^{r'} \sum_{j = l}^{i - 1} [a_i < a_j] = + \sum_{i = r + 1}^{r'} (\sum_{j = 1}^{i - 1} [a_i < a_j] - \sum_{j = 1}^{l - 1} [a_i < a_j])Δc=+∑i=r+1r′​∑j=li−1​[ai​<aj​]=+∑i=r+1r′​(∑j=1i−1​[ai​<aj​]−∑j=1l−1​[ai​<aj​])。

其中对形如 ∑i=1x−1[ax<ai]\sum_{i = 1}^{x - 1} [a_x < a_i]∑i=1x−1​[ax​<ai​] 或 ∑i=x+1n[ax>ai]\sum_{i = x + 1}^n [a_x > a_i]∑i=x+1n​[ax​>ai​] 的式子求和可以通过权值树状数组预处理 + 前缀和来做。

于是我们可以得到第二层算法要求的两种东西（共有 n32n^\frac 3 2n23​ 级别个）：

  1. A(x,p):∑i=1p[ax<ai]A(x, p) : \sum_{i = 1}^{p} [a_x < a_i]A(x,p):∑i=1p​[ax​<ai​]。
  2. B(x,p):∑i=pn[ax>ai]B(x, p) : \sum_{i = p}^n [a_x > a_i]B(x,p):∑i=pn​[ax​>ai​]。

考虑值域分块。依次插入从 111 到 nnn 中的每个数，插入 iii 后立即处理所有 p=ip = ip=i 的 AAA 问题。BBB 问题同理即可。

值得一提的是，有许多 AAA 操作（BBB 同理）的 ppp 相同、xxx 连续，且我们只关心他们的答案和，可以将其压缩存储，空间复杂度可以从 O(n32)\mathcal O(n^\frac 3 2)O(n23​) 优化为 O(n)\mathcal O(n)O(n)。

时间复杂度 O(n32)\mathcal O(n^\frac 3 2)O(n23​)，空间复杂度 O(n)\mathcal O(n)O(n)。

# include <cstdio>
# include <cstring>
# include <algorithm>

namespace Main {
    namespace Source {
        typedef short signed int hd;
        typedef short unsigned int hu;
        typedef unsigned int uint;
        typedef long long unsigned int llu;
        template <const uint N>
        class Fent {
            static inline const uint lowbit(const uint a) { return a bitand ~a + 1; }
            uint tree[N + 1];
        public:
            inline const void clear() { memset(tree, 0, sizeof tree); }
            Fent() { clear(); }
            inline const void insert(uint a) { while (a <= N) ++tree[a], a += lowbit(a); }
            inline const uint query(uint a)
            { uint r(0); while (a) r += tree[a], a ^= lowbit(a); return r; }
            inline const uint query(uint l, uint r) { return query(r) - query(l - 1); }
        };
    }
    using namespace Source;
    enum {
        N = (const uint)1e5, M = (const uint)1e5,
        S = (const hu)316.22776601683793319988935444327
    };
    static uint n, m;
    static uint a[N + 1], t[N + 1];
    namespace Mo_Algo {
        struct query {
            uint l, r;
            uint time;
            llu c;
            uint d0, d1;
            hd a0, a1;
            query() : l(1), r(1), c(), a0(0), a1(0) {}
            inline const void read(const uint time) { scanf("%u%u", &l, &r), this->time = time; }
            inline const void write() { printf("%llu\n", c); }
            friend inline const bool operator<(const query& a, const query& b)
            { return a.time < b.time; }
            static inline const bool comp(const query& a, const query& b)
            { return a.l / S == b.l / S ? a.r < b.r : a.l < b.l; }
        };
        static uint a[N + 1];
        static query q[M + 1];
        static uint sc[2][N + 1];
        namespace Block {
            struct query {
                uint l, r, p;
                uint time;
                llu c;
                query() {}
                query(const uint l, const uint r, const uint p, const uint time)
                : l(l), r(r), p(p), time(time), c() {}
                friend inline const bool operator<(const query& a, const query& b)
                { return a.time < b.time; }
                static inline const bool cmpl(const query& a, const query& b) { return a.p < b.p; }
                static inline const bool cmpg(const query& a, const query& b) { return a.p > b.p; }
            } static q[2][M * 2];
            static uint s[2];
            static inline const uint addA(const uint l, const uint r, const uint p)
            { q[0][s[0]] = query(l, r, p, s[0]); return s[0]++; }
            static inline const uint addB(const uint l, const uint r, const uint p)
            { q[1][s[1]] = query(l, r, p, s[1]); return s[1]++; }
            static inline const llu queryA(const uint time) { return q[0][time].c; }
            static inline const llu queryB(const uint time) { return q[1][time].c; }
            static inline const void solve() {
                std::sort(q[0], q[0] + s[0], query::cmpl);
                std::sort(q[1], q[1] + s[1], query::cmpg);
                static uint sc[S + 1][S], scb[S + 1];
                memset(sc, 0, sizeof sc), memset(scb, 0, sizeof scb);
                for (register uint i(1), j(0); i <= n; ++i) {
                    for (hu k(0); k < a[i] % S; ++k) ++sc[a[i] / S][k];
                    for (hu k(0); k < a[i] / S; ++k) ++scb[k];
                    while (j < s[0] and q[0][j].p < i) ++j;
                    while (j < s[0] and q[0][j].p == i) {
                        for (uint k(q[0][j].l); k <= q[0][j].r; ++k)
                            q[0][j].c += sc[a[k] / S][a[k] % S] + scb[a[k] / S];
                        ++j;
                    }
                }
                memset(sc, 0, sizeof sc), memset(scb, 0, sizeof scb);
                for (register uint i(n), j(0); i >= 1; --i) {
                    for (hu k(a[i] % S + 1); k < S; ++k) ++sc[a[i] / S][k];
                    for (hu k(a[i] / S + 1); k <= S; ++k) ++scb[k];
                    while (j < s[1] and q[1][j].p > i) ++j;
                    while (j < s[1] and q[1][j].p == i) {
                        for (uint k(q[1][j].l); k <= q[1][j].r; ++k)
                            q[1][j].c += sc[a[k] / S][a[k] % S] + scb[a[k] / S];
                        ++j;
                    }
                }
                std::sort(q[0], q[0] + s[0]), std::sort(q[1], q[1] + s[1]);
            }
        }
        static inline const void begin() {
            static Fent<N> t;
            t.clear();
            for (register uint i(1); i <= n; ++i) sc[0][i] += t.query(a[i] + 1, n), t.insert(a[i]);
            t.clear();
            for (register uint i(n); i >= 1; --i) sc[1][i] += t.query(1, a[i] - 1), t.insert(a[i]);
            for (register uint i(1); i <= n; ++i)
                sc[0][i] += sc[0][i - 1], sc[1][i] += sc[1][i - 1];
            std::sort(q + 1, q + m + 1, query::comp);
        }
        static inline const void end() {
            for (register uint i(1); i <= m; ++i)
                q[i].c += q[i - 1].c
                + (q[i].a0 ? q[i].a0 * Block::queryB(q[i].d0) : 0)
                + (q[i].a1 ? q[i].a1 * Block::queryA(q[i].d1) : 0);
            std::sort(q + 1, q + m + 1);
        }
        static inline const void solve() {
            for (register uint i(1); i <= m; ++i) {
                uint l(q[i - 1].l), r(q[i - 1].r);
                if (q[i].l < l)
                    q[i].c += sc[1][l - 1] - sc[1][q[i].l - 1],
                    q[i].d0 = Block::addB(q[i].l, l - 1, r + 1), q[i].a0 = -1, l = q[i].l;
                if (q[i].r > r)
                    q[i].c += sc[0][q[i].r] - sc[0][r],
                    q[i].d1 = Block::addA(r + 1, q[i].r, l - 1), q[i].a1 = -1, r = q[i].r;
                if (q[i].l > l)
                    q[i].c -= sc[1][q[i].l - 1] - sc[1][l - 1],
                    q[i].d0 = Block::addB(l, q[i].l - 1, r + 1), q[i].a0 = +1, l = q[i].l;
                if (q[i].r < r)
                    q[i].c -= sc[0][r] - sc[0][q[i].r],
                    q[i].d1 = Block::addA(q[i].r + 1, r, l - 1), q[i].a1 = +1, r = q[i].r;
            }
            Block::solve();
        }
        static inline const void main() { begin(); solve(); end(); }
    }
    static inline const void main() {
        scanf("%u%u", &n, &m);
        for (register uint i(1); i <= n; ++i) scanf("%u", &a[i]), t[i] = a[i];
        std::sort(t + 1, t + n + 1);
        for (register uint i(1); i <= n; ++i)
            Mo_Algo::a[i] = std::lower_bound(t + 1, t + n + 1, a[i]) - t;
        for (register uint i(1); i <= m; ++i) Mo_Algo::q[i].read(i);
        Mo_Algo::main();
        for (register uint i(1); i <= m; ++i) Mo_Algo::q[i].write();
    }
}

signed int main() { Main::main(); return 0; }
