namespace MX {
    namespace IO {
        typedef long long signed int LLD;
        typedef long long unsigned int LLU;
        inline const void space() { putchar(' '); }
        inline const void enter() { putchar('\n'); }
        inline const void dot() { putchar('.'); }
        inline const void nothing() {}

        template<typename T>
        inline T read() {
            signed char t;
            short signed int m(1);
            while (not isdigit(t = getchar())) if (t == '-') m = -m;
            T r(0);
            while (r = r * 10 + (t - '0') * m, isdigit(t = getchar()));
            if (t == '.') {
                T base(1);
                while (isdigit(t = getchar())) r += (t - '0') * (base /= 10) * m;
            }
            return r;
        }
        template<typename T>
        inline T read(T& a) {
            a = read<T>();
            return a;
        }

        template<typename T>
        inline const void write(T a, const void (*space)() = nothing) {
            if (a < 0) putchar('-'), a = -a;
            struct Point {
                signed char value;
                Point* last;
                Point() {
                    last = NULL;
                }
                Point(Point* const last) {
                    this->last = last;
                }
            };
            Point *begin(new Point(Point())), *end(begin);
            while (end->value = a % 10 + '0', end = new Point(Point(end)), a /= 10);
            while (end not_eq begin) {
                Point* const t(end->last);
                delete end, putchar((end = t)->value);
            }
            delete begin;
            space();
        }
        template<typename T>
        inline const void writef(T a, const LLU len = 6, const void (*space)() = nothing) {
            if (a < 0) putchar('-'), a = -a;
            write((const LLU)a, dot);
            for (register LLU i(0); i < len; ++i)
                write((const LLU)(a *= 10) % 10);
            space();
        }
        template<typename T>
        inline const void write(const void (*space)(), const LLU count, ...) {
            va_list list;
            va_start(list, count);
            for (register LLU i(0); i < count; ++i) write(va_arg(list, T), space);
            va_end(list);
        }
        template<typename T>
        inline const void writef(const void (*space)(), const LLU len,  const LLU count, ...) {
            va_list list;
            va_start(list, count);
            for (register LLU i(0); i < count; ++i) writef(va_arg(list, T), len, space);
            va_end(list);
        }
    }
