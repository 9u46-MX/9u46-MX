P5174.
Luogu blog.

4∑i2+j2⩽r(i2+j2).4\sum_{i^2 + j^2 \leqslant r}(i^2 + j^2). 4i2+j2⩽r∑​(i2+j2).

考虑枚举 iii，然后发现 jjj 能取到的最大值单调递减。故可以 O(n)\mathcal O(n)O(n) 枚举。设对于 iii 最大的 jjj 为 cic_ici​，则：

4∑i2+j2⩽r(i2+j2)=∑i=1⌊r⌋∑j=1ci(i2+j2)=∑i=1⌊r⌋(ci⋅i2+∑j=1cij2)=∑i=1⌊r⌋(ci⋅i2+ci(ci+1)(2ci+1)6).\begin{aligned} &4\sum_{i^2 + j^2 \leqslant r}(i^2 + j^2)\\ =&\sum_{i = 1}^{\lfloor \sqrt r \rfloor} \sum_{j = 1}^{c_i} (i^2 + j^2)\\ =&\sum_{i = 1}^{\lfloor \sqrt r \rfloor} (c_i \cdot i^2 + \sum_{j = 1}^{c_i} j^2)\\ =&\sum_{i = 1}^{\lfloor \sqrt r \rfloor} (c_i \cdot i^2 + \frac{c_i(c_i + 1)(2c_i + 1)}6).\\ \end{aligned} ===​4i2+j2⩽r∑​(i2+j2)i=1∑⌊r​⌋​j=1∑ci​​(i2+j2)i=1∑⌊r​⌋​(ci​⋅i2+j=1∑ci​​j2)i=1∑⌊r​⌋​(ci​⋅i2+6ci​(ci​+1)(2ci​+1)​).​

时空复杂度 O(r)\mathcal O(\sqrt r)O(r​)。

代码中 jjj 枚举的是 cic_ici​。

# include <cstdio>

namespace Main {
    namespace Source {
        typedef unsigned int uint;
        typedef long long unsigned int llu;
        namespace Maths {
            static inline const uint qpow(uint b, uint e, const uint mod) {
                uint r(1);
                while (e) {
                    if (e bitand 1) r = 1ull * r * b % mod;
                    b = 1ull * b * b % mod, e >>= 1;
                }
                return r;
            }
            static inline const uint inv(const uint x, const uint p) { return qpow(x, p - 2, p); }
            static inline const uint frac(const uint a, const uint b, const uint p)
            { return 1ull * a * inv(b, p) % p; }
        }
    }
    using namespace Source;
    enum { Sqrt = (const uint)1e7l };
    static const uint Mod(1e9 + 7), F16(166666668); // frac 1/6 mod (10^9 + 7) = 166666668
    static llu r;
    static uint sqrt;
    static uint ans;
    // static uint sumsqr[Sqrt + 1];
    static inline const void main() {
        // printf("%u\n", Maths::frac(1, 6, Mod));
        scanf("%llu", &r);
        while (1ull * sqrt * sqrt <= r) ++sqrt; --sqrt;
        // for (register uint i(1); i <= sqrt; ++i) sumsqr[i] = (sumsqr[i - 1] + 1ull * i * i) % Mod;
        for (register uint i(1), j(sqrt); i <= sqrt; ++i) {
            while (1ull * i * i + 1ull * j * j > r) --j;
            ans = (ans +
                1ull * i * i % Mod * (j + 1) % Mod
                + 1ull * j % Mod * (j + 1) % Mod * (2 * j + 1) % Mod * F16 % Mod
            ) % Mod;
        }
        printf("%u\n", ans = (ans * 4) % Mod);
    }
}

signed int main() { Main::main(); return 0; }
