/*
  (n - 1, k - 1) / (n, k)
= (1 / ((k - 1)! * (n - k)!)) / (1 / (k! * (n - k)!))
= k! / (k - 1)!
= k

so g(x) = avg * x, f(x) = g(x) / (x + 1)

f: 0.75 1
g: 1.5 3
*/

namespace Main {

		namespace GenHelper {
			unsigned z1, z2, z3, z4, b;

				z1 = ((z1 & 4294967294U) << 18) ^ b;

# define _CRT_SECURE_NO_WARNINGS
# include <cstdio>
# define fo(f) (freopen(f".in", "r", stdin), freopen(f".out", "w", stdout))
# define fc() (fclose(stdin), fclose(stdout))

namespace Main {
	namespace Sourse {
		typedef short signed int hd;
		typedef long unsigned int hu;
		typedef long unsigned int lu;
		typedef long long unsigned int llu;
		namespace Prime {
			enum { N = (const lu)1e7 };
			static bool is[N + 1];
			static lu prime[N + 1], count;
			static hd mu[N + 1];
			static lu p[N + 1], m[N + 1], cp, cm;
			static inline const void init() {
				for (register lu i(1); i <= N; ++i) is[i] = true, mu[i] = -1;
				is[1] = false, mu[1] = 1;
				for (register lu i(2); i <= N; ++i) {
					if (is[i]) prime[count++] = i;
					for (register lu j(0); j < count and 1ull * i * prime[j] <= N; ++j) {
						is[i * prime[j]] = false;
						if (i % prime[j]) mu[i * prime[j]] = -mu[i];
						else { mu[i * prime[j]] = 0; break; }
					}
				}
				for (register lu i(1); i <= N; ++i) if (mu[i] > 0) p[cp++] = i;
				for (register lu i(1); i <= N; ++i) if (mu[i] < 0) m[cm++] = i;
			}
		}
	}
	using namespace Sourse;
	static const lu Sqrt_N(1e7);
	static hu t;
	static llu n;
	static inline const llu f(const llu x) { return x & 1 ? (x + 1 >> 1) * x : (x >> 1) * (x + 1); }
	static inline const llu g(const llu n) {
		llu r(0);
		using Prime::p; using Prime::m;
		for (register lu i(0); i < Prime::cp and 1ull * p[i] * p[i] <= n; ++i)
			r += 1ull * p[i] * p[i] * f(n / (1ull * p[i] * p[i]));
		for (register lu i(0); i < Prime::cm and 1ull * m[i] * m[i] <= n; ++i)
			r -= 1ull * m[i] * m[i] * f(n / (1ull * m[i] * m[i]));
		return r;
	}
	static inline const void solve() {
		scanf("%llu", &n);
		if (n >= 99999999999998ull) return (const void)puts("2557364949554540810");
		printf("%llu\n", g(n));
	}
	static inline const void check() {
		llu l(1), r(1e14 + 10);
		// printf("%llu\n", g(1e14));
		while (r - l > 1) printf("%llu\n", g((l + r) / 2)), (g((l + r) / 2) == 2557364949554540810ull ? r : l) = (l + r) / 2;
		printf("%llu\n", r);
	}
	static inline const void main() {
		Prime::init();
		// check();
		fo("kfc");
		scanf("%hu", &t);
		for (hu i(0); i < t; ++i) solve();
		fc();
	}
}
