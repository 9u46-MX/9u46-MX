A - escape

# define _CRT_SECURE_NO_WARNINGS
# include <cstdio>
# define fo(file) (freopen(file".in", "r", stdin), freopen(file".out", "w", stdout))
# define fc()     (fclose(stdin),                  fclose(stdout)                  )

namespace Main {
	namespace Sourse {
		typedef long signed int ld;
		typedef long unsigned int lu;
		typedef long long unsigned int llu;
		template<typename T> static inline const void amin(T& a, const T& b) { if (b < a) a = b; }
		template<const lu Size>
		class Fenwick_tree {
			ld tree[Size + 1];
		public:
			Fenwick_tree() { for (register lu i(1); i <= Size; ++i) tree[i] = lowbit(i); }
			static inline const lu lowbit(const lu x) { return x bitand ~x + 1; }
			inline const void modify(lu x) { ++x; while (x <= Size) --tree[x], x += lowbit(x); }
			inline const lu query(lu x) const { ++x; ld r(0); while (x) r += tree[x], x ^= lowbit(x); return r; }
		};
	}
	using namespace Sourse;
	enum { N = 500000, M = 26 };
	static lu n;
	static llu k;
	static char s[N + 1], ans[N + 1];
	static lu p[M][N + 1], l[M], h[M];
	static Fenwick_tree<N> t;
	static inline const void main() {
		fo("escape");
		scanf("%lu%llu", &n, &k);
		scanf("%s", s);
		for (register lu i(0); i < n; ++i) p[s[i] - 'a'][l[s[i] - 'a']++] = i;
		for (register lu i(0), m(0); i < n; ++i) {
			char u(0);
			for (char j(0); j < M; ++j) {
				// printf("%c %lu %lu %lu %lu\n", j, h[j], l[j], t.query(p[j][h[j]]), k + 1);
				if (h[j] < l[j] and t.query(p[j][h[j]]) <= k + 1)
				{ u = 'a' + j, s[p[j][h[j]]] = 0, t.modify(p[j][h[j]]), k -= t.query(p[j][h[j]]), ++h[j]; break; }
			}
			if (not u) { /*printf("%lu\n", i);*/ while (not s[m]) ++m; u = s[m], s[m] = 0; }
			ans[i] = u;
		}
		puts(ans);
		fc();
	}
}

signed int main() { Main::main(); return 0; }

B - ride

# define _CRT_SECURE_NO_WARNINGS
# include <cstdio>
# include <vector>
# define fo(file) (freopen(file".in", "r", stdin), freopen(file".out", "w", stdout))
# define fc()     (fclose(stdin),                  fclose(stdout)                  )

namespace Main {
	namespace Sourse {
		typedef long unsigned int lu;
		typedef long long unsigned int llu;
		typedef std::vector<lu> vector;
	}
	using namespace Sourse;
	enum { N = 300000 };
	class tree {
		struct node {
			 vector out;
			 lu lower;
			 node() : out(), lower(1) {}
			 inline const void add(const lu to) { out.push_back(to); }
		} o[N + 1];
	public:
		tree() { for (register lu i(1); i <= N; ++i) o[i] = node(); };
		inline const void add(const lu u, const lu v) { o[u].add(v), o[v].add(u); }
		inline const void search_subtree(const lu x = 1, const lu m = 0) {
			for (register lu i(0); i < o[x].out.size(); ++i) {
				if (o[x].out[i] not_eq m) {
					search_subtree(o[x].out[i], x);
					if (o[x].out[i] < x) o[x].lower += o[o[x].out[i]].lower;
				}
			}
			// printf("%lu %lu %lu\n", x, o[x].lower, o[x].upper);
		}
		inline const llu search_answer(const lu x = 1, const lu m = 0, lu lower = 0) {
			llu r;
			if (m < x) r = 1ull * lower * (o[x].lower - 1), o[x].lower += lower;
			else r = lower = 0;
			// printf("%lu %lu %lu %llu\n", x, m, o[x].lower, r);
			lower = o[x].lower;
			for (register lu i(0); i < o[x].out.size(); ++i) {
				if (o[x].out[i] not_eq m) {
					if (o[x].out[i] < x)
						r +=
						1ull * o[o[x].out[i]].lower * (o[x].lower - o[o[x].out[i]].lower - 1);
					r += search_answer(o[x].out[i], x, o[x].out[i] < x ? 0 : lower);
				}
			}
			return r;
		}
	};
	static lu n;
	static tree t;
	static inline const void main() {
		fo("ride");
		scanf("%lu", &n);
		for (register lu i(1); i < n; ++i) { static lu u, v; scanf("%lu%lu", &u, &v); t.add(u, v); }
		t.search_subtree();
		// t.search_goddess();
		printf("%llu\n", t.search_answer());
		fc();
	}
}

signed int main() { Main::main(); return 0; }

C - rode

# define _CRT_SECURE_NO_WARNINGS
# include <cstdio>
# include <queue>
# include <utility>
# include <vector>
# define fo(file) (freopen(file".in", "r", stdin), freopen(file".out", "w", stdout))
# define fc()     (fclose(stdin),                  fclose(stdout)                  )

namespace Main {
	namespace Sourse {
		typedef short unsigned int hu;
		typedef unsigned int u;
		typedef std::vector<hu> vector;
		typedef std::pair<hu, hu> pair;
		typedef std::queue<pair> queue;
	}
	using namespace Sourse;
	enum { N = 5000 };
	class graph {
		struct vertex {
			vector out;
			hu f[N + 1][2];
			vertex() : out() { for (register hu i(1); i <= N; ++i) f[i][1] = f[i][0] = N; }
			inline const void add(const hu to) { out.push_back(to); }
		} v[N + 1];
		inline const void search(const hu s) {
			queue q; q.push(pair(s, v[s].f[s][0] = 0));
			while (not q.empty()) {
				const pair t(q.front()); q.pop();
				for (register hu i(0); i < v[t.first].out.size(); ++i)
					if (v[s].f[v[t.first].out[i]][(t.second + 1) % 2] == N)
						q.push(pair(v[t.first].out[i], v[s].f[v[t.first].out[i]][(t.second + 1) % 2] = t.second + 1));
			}
		}
	public:
		graph() { for (register hu i(1); i <= N; ++i) v[i] = vertex(); }
		inline const void add(const hu x, const hu y) { v[x].add(y), v[y].add(x); }
		inline const void init() { for (register hu i(1); i <= N; ++i) search(i); }
		inline const bool query(const hu s, const hu t, const u d) {
			return s == t && v[s].out.empty() ? !d : v[s].f[t][d % 2] < N && v[s].f[t][d % 2] <= d; }
	};
	static hu n, m;
	static u q;
	static graph g;
	static inline const void main() {
		fo("road");
		scanf("%hu%hu%u", &n, &m, &q);
		for (register hu i(0); i < m; ++i) { static hu x, y; scanf("%hu%hu", &x, &y); g.add(x, y); }
		g.init();
		for (register u i(0); i < q; ++i)
		{ static hu s, t; static u d; scanf("%hu%hu%u", &s, &t, &d); puts(g.query(s, t, d) ? "Yes" : "No"); }
		fc();
	}
}

signed int main() { Main::main(); return 0; }

D - tree

# define _CRT_SECURE_NO_WARNINGS
# include <cstdio>
# include <vector>
# define fo(file) (freopen(file".in", "r", stdin), freopen(file".out", "w", stdout))
# define fc()     (fclose(stdin),                  fclose(stdout)                  )

namespace Main {
	namespace Sourse {
		typedef long unsigned int lu;
		typedef std::vector<lu> vector;
		namespace Maths {
			static inline const lu qpow(lu b, lu e, const lu mod) {
				lu r(1);
				while (e) {
					if (e bitand 1) r = 1ull * r * b % mod;
					b = 1ull * b * b % mod, e >>= 1;
				}
				return r;
			}
			static inline const lu inv_prime(const lu x, const lu p) { return qpow(x, p - 2, p); }
			static inline const lu P(const lu n, const lu mod) {
				lu r(1);
				for (register lu i(n / 2 + 1); i <= n; ++i) r = 1ull * r * i % mod;
				r = 1ull * r * inv_prime(qpow(2, n / 2, mod), mod) % mod;
				return r;
			}
		}
	}
	using namespace Sourse;
	enum { N = 300000 };
	static const lu Mod(998244353);
	static const lu inv2(Maths::inv_prime(2, Mod));
	static lu P2, P4;
	class tree {
		struct node {
			vector out;
			lu sum0, sum1, sum2;
			node() : out(), sum0(1), sum1(0), sum2(0) {}
			inline const void add(const lu to) { out.push_back(to); }
		} o[N + 1];
		lu s;
		lu a, b, c;
		inline const void search_s(const lu x = 1, const lu m = 0) {
			s = (s + 1ull * o[x].sum1 * inv2) % Mod;
			for (lu i(0); i < o[x].out.size(); ++i) if (o[x].out[i] not_eq m) search_s(o[x].out[i], x);
		}
		inline const void search_a(const lu x = 1, const lu m = 0) {
			a = (a + 1ull * inv2 % Mod * (P2 + P4) % Mod * o[x].sum2 % Mod) % Mod;
			for (lu i(0); i < o[x].out.size(); ++i) if (o[x].out[i] not_eq m) search_a(o[x].out[i], x);
		}
		inline const void search_b(const lu x = 1, const lu m = 0) { b = 1ull * P4 % Mod * s % Mod * s % Mod; }
		inline const void search_c(const lu x = 1, const lu m = 0) {
			c = (c + 1ull * P4 % Mod * o[x].sum1 % Mod * o[x].sum1 % Mod) % Mod;
			for (lu i(0); i < o[x].out.size(); ++i) if (o[x].out[i] not_eq m) search_c(o[x].out[i], x);
		}
	public:
		tree() { for (register lu i(1); i <= N; ++i) o[i] = node(); }
		inline const void add(const lu u, const lu v) { o[u].add(v), o[v].add(u); }
		inline const void search_subtree(const lu x = 1, const lu m = 0) {
			for (lu i(0); i < o[x].out.size(); ++i)
				if (o[x].out[i] not_eq m)
					search_subtree(o[x].out[i], x),
					o[x].sum0 += o[o[x].out[i]].sum0,
					o[x].sum1 = (o[x].sum1 + o[o[x].out[i]].sum1 + o[o[x].out[i]].sum0) % Mod,
					o[x].sum2 = (o[x].sum2 + o[o[x].out[i]].sum2 + 2 * o[o[x].out[i]].sum1 + o[o[x].out[i]].sum0) % Mod;
			// putchar('\n'),
			// printf("d^0(%lu) = %lu\n", x, o[x].sum0),
			// printf("d^1(%lu) = %lu\n", x, o[x].sum1),
			// printf("d^2(%lu) = %lu\n", x, o[x].sum2),
			// putchar('\n');
		}
		inline const void search_goddess(const lu x = 1, const lu m = 0, lu sum0 = 0, lu sum1 = 0, lu sum2 = 0) {
			sum0 = o[x].sum0 += sum0, sum1 = o[x].sum1 = (o[x].sum1 + sum1) % Mod, sum2 = o[x].sum2 = (o[x].sum2 + sum2) % Mod;
			// putchar('\n'),
			// printf("d^0(%lu) = %lu\n", x, o[x].sum0),
			// printf("d^1(%lu) = %lu\n", x, o[x].sum1),
			// printf("d^2(%lu) = %lu\n", x, o[x].sum2),
			// putchar('\n');
			for (lu i(0); i < o[x].out.size(); ++i) {
				if (o[x].out[i] not_eq m) {
					const lu _sum0(sum0 - o[o[x].out[i]].sum0);
					const lu _sum1((sum1 + Mod - o[o[x].out[i]].sum1 + Mod - o[o[x].out[i]].sum0 + _sum0) % Mod);
					const lu _sum2((
							0ull + sum2
							+ Mod - o[o[x].out[i]].sum2
							+ Mod - 2 * o[o[x].out[i]].sum1 % Mod
							+ Mod - o[o[x].out[i]].sum0
							+ 2 * _sum1 - _sum0
					) % Mod);
					search_goddess(o[x].out[i], x, _sum0, _sum1, _sum2);
				}
			}
		}
		inline const lu search_answer(const lu x = 1, const lu m = 0) {
			search_s(), search_a(), search_b(), search_c();
			// printf("%lu %lu %lu\n", a, b, c);
			return (a + b + Mod - c) % Mod;
		}
	};
	static lu n;
	static tree t;
	static inline const void begin() {
		fo("tree");
		scanf("%lu", &n), P2 = Maths::P(n - 2, Mod), P4 = Maths::P(n - 4, Mod);
		// printf("%lu\n", );
		for (register lu i(1); i < n; ++i) { static lu u, v; scanf("%lu%lu", &u, &v); t.add(u, v); }
	}
	static inline const void solve() { t.search_subtree(), t.search_goddess(); }
	static inline const void end() { printf("%lu\n", t.search_answer() % Mod); fc(); }
	static inline const void main() { begin(), solve(), end(); }
}

signed int main() { Main::main(); return 0; }
